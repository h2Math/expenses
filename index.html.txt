            const year = date.getFullYear();

        function calculateEarnings(student, start, end) {
            const startDate = new Date(start);
            const endDate = new Date(end);
            const hours = (endDate - startDate) / (1000 * 60 * 60);
            if (student.rateType === 'hourly') {
                return hours * student.rateAmount;
            } else {
                return student.rateAmount;
            }

        function exportToFile() {
            const data = getAllData();
            const dataStr = JSON.stringify(data, null, 2);
            const blob = new Blob([dataStr], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const link = document.createElement('a');
            link.href = url;
            link.download = `tuition_backup_${new Date().toISOString().split('T')[0]}.json`;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            URL.revokeObjectURL(url);
            showMessage('✅ Data exported successfully! Save the file to your preferred location.', 'success', 'backupFormMessage');
        }

        function importFromFile() {
            const fileInput = document.createElement('input');
            fileInput.type = 'file';
            fileInput.accept = '.json';
            fileInput.onchange = function(event) {
                const file = event.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        try {
                            const backupData = JSON.parse(e.target.result);
                            
                            if (backupData.groups && backupData.students && backupData.lessons) {
                                if (confirm('⚠️ WARNING: This will replace ALL current data. Continue?')) {
                                    groups = backupData.groups;
                                    students = backupData.students;
                                    lessons = backupData.lessons;
                                    
                                    saveGroups();
                                    saveStudents();
                                    saveLessons();
                                    
                                    // Refresh all views
                                    renderGroups();
                                    renderStudents();
                                    renderLessons();
                                    updateLessonStats();
                                    updateCalendarStats();
                                    updateBackupStats();
                                    updateSelectors();
                                    
                                    showMessage('✅ Data restored successfully!', 'success', 'backupFormMessage');
                                }
                            } else {
                                showMessage('❌ Invalid backup file format', 'error', 'backupFormMessage');
                            }
                        } catch (error) {
                            showMessage('❌ Error parsing backup file', 'error', 'backupFormMessage');
                        }
                    };
                    reader.readAsText(file);
                }
            };
            fileInput.click();
        }

            if (new Date(start) >= new Date(end)) {
                showMessage('End time must be after start time', 'error', 'lessonFormMessage');
                return;
            }
            
            const student = students.find(s => s.id === studentId);
            const startDate = new Date(start);
            const endDate = new Date(end);
            const hours = (endDate - startDate) / (1000 * 60 * 60);
            const earnings = calculateEarnings(student, start, end);
            
            const newLesson = {
                id: Date.now().toString(),
                studentId: studentId,
                studentName: student.name,
                groupId: student.groupId,
                start: start,
                end: end,
                topic: topic,
                notes: notes,
                earnings: earnings,
                hours: hours
            };
            
            lessons.push(newLesson);
            saveLessons();
            renderLessons();
            updateLessonStats();
            updateCalendarStats();
            updateBackupStats();
            
            document.getElementById('lessonTopic').value = '';
            document.getElementById('lessonNotes').value = '';
            
            showMessage(`✅ Lesson added for ${student.name} - ${hours.toFixed(1)} hours = $${earnings.toFixed(2)}`, 'success', 'lessonFormMessage');
            
            const now = new Date();
            now.setHours(now.getHours() + 1);
            now.setMinutes(0);
            document.getElementById('lessonStart').value = getDateTimeLocal(now);
            const event = new Event('change');
            document.getElementById('lessonStart').dispatchEvent(event);
        }

        function formatICSDate(date) {
            return date.toISOString().replace(/[-:]/g, '').split('.')[0] + 'Z';
        }

        function escapeICS(str) {
            if (!str) return '';
            return str.replace(/[\\;,\n]/g, function(match) {
                if (match === '\\') return '\\\\';
                if (match === ';') return '\\;';
                if (match === ',') return '\\,';
                if (match === '\n') return '\\n';
                return match;
            });
        }

        function generateICSForAllLessons() {
            let icsContent = `BEGIN:VCALENDAR
VERSION:2.0
PRODID:-//Mr Lee Math Tuition//EN
CALSCALE:GREGORIAN
METHOD:PUBLISH
X-WR-CALNAME:Tuition Schedule
X-WR-CALDESC:Tuition lessons schedule
`;
            
            for (const lesson of lessons) {
                const startDate = new Date(lesson.start);
                const endDate = new Date(lesson.end);
                const uid = `${lesson.id}@mrleemath.com`;
                const summary = lesson.studentName;
                const description = lesson.notes || '';
                
                icsContent += `BEGIN:VEVENT
UID:${uid}
DTSTAMP:${formatICSDate(new Date())}
DTSTART:${formatICSDate(startDate)}
DTEND:${formatICSDate(endDate)}
SUMMARY:${escapeICS(summary)}
DESCRIPTION:${escapeICS(description)}
CLASS:PUBLIC
END:VEVENT
`;
            }
            
            icsContent += `END:VCALENDAR`;
            return icsContent;
        }

        async function generateAnalysis() {
            const fromDate = document.getElementById('analysisFromDate').value;
            const toDate = document.getElementById('analysisToDate').value;
            
            if (!fromDate || !toDate) {
                showMessage('Please select a date range', 'error', 'analysisFormMessage');
                return;
            }
            
            const filteredLessons = lessons.filter(l => {
                const lessonDate = new Date(l.start.split('T')[0]);
                return lessonDate >= new Date(fromDate) && lessonDate <= new Date(toDate);
            });
            
            if (filteredLessons.length === 0) {
                showMessage('No lessons found in this date range', 'error', 'analysisFormMessage');
                return;
            }
            
            const groupTotals = {};
            const groupPaymentStatus = {};
            const studentTotals = {};
            let overallTotal = 0;
            let totalHours = 0;
            
            for (const lesson of filteredLessons) {
                const group = groups.find(g => g.id === lesson.groupId);
                const groupName = group?.name || 'Unknown Group';
                const studentName = lesson.studentName;
                
                groupTotals[groupName] = (groupTotals[groupName] || 0) + lesson.earnings;
                groupPaymentStatus[groupName] = group?.paymentStatus || 'unpaid';
                studentTotals[studentName] = {
                    group: groupName,
                    earnings: (studentTotals[studentName]?.earnings || 0) + lesson.earnings,
                    hours: (studentTotals[studentName]?.hours || 0) + lesson.hours,
                    lessons: (studentTotals[studentName]?.lessons || 0) + 1
                };
                overallTotal += lesson.earnings;
                totalHours += lesson.hours;
            }
            
            let analysisHTML = `
                <div id="analysisContainer" class="analysis-container" style="background: white; color: #1a1a2e; padding: 30px; font-family: Arial, sans-serif; border-radius: 20px; max-width: 800px; margin: 0 auto;">
                    <div class="analysis-header">
                        <div class="analysis-title">FEE ANALYSIS & PAYMENT REPORT</div>
                        <div class="analysis-dates">Period: ${formatDateForDisplay(fromDate)} to ${formatDateForDisplay(toDate)}</div>
                    </div>
                    
                    <div class="summary-card">
                        <div class="summary-number">$${overallTotal.toFixed(2)}</div>
                        <div class="summary-label">Total Fees Earned</div>
                        <div style="margin-top: 10px;">📊 ${totalHours.toFixed(1)} total hours | 📚 ${filteredLessons.length} total lessons</div>
                    </div>
                    
                    <h3 style="margin: 20px 0 10px 0;">By Group (with Payment Status)</h3>
                    <table class="analysis-table">
                        <thead>
                            <tr><th>Group</th><th>Total Fees</th><th>% of Total</th><th>Payment Status</th></tr>
                        </thead>
                        <tbody>
            `;
            
            for (const [groupName, total] of Object.entries(groupTotals).sort((a, b) => b[1] - a[1])) {
                const percentage = (total / overallTotal * 100).toFixed(1);
                const status = groupPaymentStatus[groupName];
                const statusClass = status === 'paid' ? 'status-paid' : (status === 'partial' ? 'status-partial' : 'status-unpaid');
                const statusText = status === 'paid' ? '✅ Paid' : (status === 'partial' ? '🔄 Partial' : '⚠️ Unpaid');
                analysisHTML += `
                    <tr>
                        <td>${escapeHtml(groupName)}</td>
                        <td>$${total.toFixed(2)}</td>
                        <td>${percentage}%</td>
                        <td><span class="payment-status-badge ${statusClass}">${statusText}</span></td>
                    </tr>
                `;
            }
            
            analysisHTML += `
                        </tbody>
                    </table>
                    
                    <h3 style="margin: 20px 0 10px 0;">By Student</h3>
                    <table class="analysis-table">
                        <thead>
                            <tr><th>Student</th><th>Group</th><th>Hours</th><th>Lessons</th><th>Total Fees</th></tr>
                        </thead>
                        <tbody>
            `;
            
            for (const [studentName, data] of Object.entries(studentTotals).sort((a, b) => b[1].earnings - a[1].earnings)) {
                analysisHTML += `
                    <tr>
                        <td>${escapeHtml(studentName)}</td>
                        <td>${escapeHtml(data.group)}</td>
                        <td>${data.hours.toFixed(1)}</td>
                        <td>${data.lessons}</td>
                        <td>$${data.earnings.toFixed(2)}</td>
                    </tr>
                `;
            }
            
            analysisHTML += `
                        </tbody>
                    </table>
                    
                    <div class="analysis-total">
                        <p><strong>GRAND TOTAL: $${overallTotal.toFixed(2)}</strong></p>
                    </div>
                    <div style="margin-top: 20px; font-size: 0.7rem; text-align: center; color: #666;">
                        Generated on ${new Date().toLocaleDateString()}
                    </div>
                </div>
            `;
            
            document.getElementById('analysisPreview').innerHTML = analysisHTML;
            document.getElementById('analysisPreview').style.display = 'block';
            document.getElementById('shareAnalysisBtn').style.display = 'block';
            currentAnalysisElement = document.getElementById('analysisContainer');
            
            showMessage('✅ Report generated! Click "Share Report" to send.', 'success', 'analysisFormMessage');
        }

        function setDefaultDates() {
            const now = new Date();
            now.setHours(now.getHours() + 1);
            now.setMinutes(0);
            const defaultStart = new Date(now);
            const defaultEnd = new Date(defaultStart.getTime() + 90 * 60 * 1000);
            
            if (!document.getElementById('lessonStart').value) {
                document.getElementById('lessonStart').value = getDateTimeLocal(defaultStart);
                document.getElementById('lessonEnd').value = getDateTimeLocal(defaultEnd);
            }
            
            const today = new Date();
            const firstDay = new Date(today.getFullYear(), today.getMonth(), 1);
            const lastDay = new Date(today.getFullYear(), today.getMonth() + 1, 0);
            if (!document.getElementById('invoiceFromDate').value) {
                document.getElementById('invoiceFromDate').value = firstDay.toISOString().split('T')[0];
                document.getElementById('invoiceToDate').value = lastDay.toISOString().split('T')[0];
            }
            if (!document.getElementById('analysisFromDate').value) {
                document.getElementById('analysisFromDate').value = firstDay.toISOString().split('T')[0];
                document.getElementById('analysisToDate').value = lastDay.toISOString().split('T')[0];
            }
        }

        function formatDate(datetimeLocal) {
            const date = new Date(datetimeLocal);
            return date.toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
        }
        
        function formatDateForDisplay(dateStr) {
            const date = new Date(dateStr);
            return date.toLocaleDateString('en-US', { year: 'numeric', month: 'short', day: 'numeric' });
        }
        
        function formatTime(datetimeLocal) {
            const date = new Date(datetimeLocal);
            return date.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' });
        }

            const msgDiv = document.getElementById(elementId);
            if (msgDiv) {

                msgDiv.style.color = type === 'success' ? '#00d09c' : '#ff4d6d';

        function escapeHtml(str) {
            if (!str) return '';
            return str.replace(/[&<>]/g, m => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;' }[m] || m));
        }
